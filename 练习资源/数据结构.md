## (一)数组模拟基础数据结构

### (1.1) 单链表+邻接表

```c++
#include <bits/stdc++.h>
using namespace std;

/**
 * head 头节点的索引
 * idx  存储当前已经用到了哪个节点
 * nxt  当前节点指向的下一个节点的索引
 * val  当前节点数值
*/

#define MAXN 100005

int n, m, x, k;
int head, tail, idx, nxt[MAXN], val[MAXN];

// 初始化链表
void init(){
    head = -1, tail = -1;
    idx = 0;
}

// 插入链表的头部
void insert_head(int x){
    val[idx] = x;
    nxt[idx] = head;
    head = idx++;
}

// 插入链表的尾部
void insert_tail(int x){
    val[idx] = x;
    nxt[idx] = nxt[tail];
	nxt[tail] = idx++;
}

// 插入第K个位置后面
void insert_kth(int x, int k){
    val[idx] = x;
    nxt[idx] = nxt[k];
    nxt[k] = idx++;
}

// 移除第K个位置元素
void remove_kth(int k){
    nxt[k] = nxt[nxt[k]];
}

```
